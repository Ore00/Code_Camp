JavaScript performance can be affected when a script is written in a way that causes the browser to work harder than it has to. One way to prevent this is to only use nested loops (for/while) when required. If a loop is required and the nested loop is based on an array size, declare the array size outside of the loop, so the system doesn't have to repeat determining the array size.

User Story: The DNA strand is missing the pairing element. Take each character, get its pair, and return the results as a 2d array. Note: this example will display the 2d array results as a string

JavaScript Code: DNA Pairing

Related JS: DNAparing.js
function pairElement(str) {
var pairArray = [ {"A": "T"}, {"T": "A"},{"C": "G"},{"G":"C"}];
var myArr = [];
var pairLength = pairArray.length;
str = str.split("");
for(var char in str){
var newStr = [];
for(var x = 0; x < pairLength; x++){
var obj = pairArray[x];
if(obj.hasOwnProperty(str[char]) === true){
newStr.push(str[char]);
newStr.push(obj[str[char]]);
myArr.push(newStr);
} } }
str = myArr;
return str;
}
